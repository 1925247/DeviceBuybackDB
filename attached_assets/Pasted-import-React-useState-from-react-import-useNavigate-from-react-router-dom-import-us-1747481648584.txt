import React, { useState } from "react";
import { useNavigate } from "react-router-dom";
import { useQuery } from "@tanstack/react-query";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
import {
  Card,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from "@/components/ui/card";
import {
  ChevronRight,
  Search,
  Plus,
  Layers,
  AlertCircle,
  HelpCircle,
  MonitorSmartphone,
  Smartphone,
  Laptop,
  TabletSmartphone,
  Watch,
} from "lucide-react";
import { Badge } from "@/components/ui/badge";
import { useToast } from "@/hooks/use-toast";

interface QuestionGroup {
  id: number;
  name: string;
  statement?: string | null;
  deviceTypeId?: number | null;
  icon?: string | null;
  active: boolean;
  questionsCount?: number;
}

interface DeviceType {
  id: number;
  name: string;
  slug: string;
}

export default function AdminConditionQuestions() {
  const navigate = useNavigate();
  const [searchQuery, setSearchQuery] = useState("");
  const [activeTab, setActiveTab] = useState("all");
  const { toast } = useToast();

  // Fetch all question groups
  const { data: questionGroups, isLoading: isGroupsLoading } = useQuery<
    QuestionGroup[]
  >({
    queryKey: ["/api/question-groups"],
  });

  // Fetch device types for filtering
  const { data: deviceTypes, isLoading: isDeviceTypesLoading } = useQuery<
    DeviceType[]
  >({
    queryKey: ["/api/device-types"],
  });

  // Handle navigating to question group details
  const handleGroupClick = (groupId: number) => {
    navigate(`/admin/condition-questions/${groupId}`);
  };

  // Handle navigating to create new group
  const handleCreateGroup = () => {
    navigate("/admin/question-groups");
  };

  // Get icon for a group based on device type
  const getGroupIcon = (group: QuestionGroup) => {
    if (group.icon) {
      return <img src={group.icon} alt={group.name} className="h-10 w-10" />;
    }

    // Default icon based on device type
    if (!group.deviceTypeId) {
      return <Layers className="h-10 w-10 text-gray-500" />;
    }

    const deviceType = deviceTypes?.find((dt) => dt.id === group.deviceTypeId);

    if (!deviceType) {
      return <HelpCircle className="h-10 w-10 text-gray-500" />;
    }

    if (
      deviceType.slug.includes("smartphone") ||
      deviceType.slug.includes("mobile") ||
      deviceType.slug.includes("phone")
    ) {
      return <Smartphone className="h-10 w-10 text-blue-500" />;
    } else if (deviceType.slug.includes("laptop")) {
      return <Laptop className="h-10 w-10 text-indigo-500" />;
    } else if (deviceType.slug.includes("tablet")) {
      return <TabletSmartphone className="h-10 w-10 text-purple-500" />;
    } else if (deviceType.slug.includes("watch")) {
      return <Watch className="h-10 w-10 text-green-500" />;
    }

    return <MonitorSmartphone className="h-10 w-10 text-indigo-500" />;
  };

  // Filter question groups based on search query and active tab
  const filteredGroups = questionGroups?.filter((group) => {
    // Filter by search query
    const matchesSearch =
      group.name?.toLowerCase().includes(searchQuery.toLowerCase()) ||
      group.statement?.toLowerCase().includes(searchQuery.toLowerCase());

    // Filter by device type tab
    if (activeTab === "all") {
      return matchesSearch;
    } else {
      const deviceType = deviceTypes?.find((dt) => dt.slug === activeTab);
      return matchesSearch && group.deviceTypeId === deviceType?.id;
    }
  });

  return (
    <div className="container mx-auto py-6">
      <header className="flex flex-col md:flex-row justify-between items-start md:items-center mb-6">
        <div>
          <h1 className="text-2xl font-bold tracking-tight">Q&A Management</h1>
          <p className="text-gray-500 mt-1">
            Manage question groups and assessments for device conditions
          </p>
        </div>
        <div className="mt-4 md:mt-0 flex flex-col md:flex-row gap-4">
          <div className="relative">
            <Search className="absolute left-2.5 top-2.5 h-4 w-4 text-gray-500" />
            <Input
              type="search"
              placeholder="Search question groups..."
              className="pl-8 w-full md:w-[250px]"
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
            />
          </div>
          <Button onClick={handleCreateGroup}>
            <Plus className="mr-2 h-4 w-4" /> Create Group
          </Button>
        </div>
      </header>

      {/* Device Type Tabs */}
      <Tabs
        defaultValue="all"
        className="mb-6"
        value={activeTab}
        onValueChange={setActiveTab}
      >
        <TabsList className="mb-2">
          <TabsTrigger value="all">All Groups</TabsTrigger>
          {deviceTypes?.map((deviceType) => (
            <TabsTrigger key={deviceType.id} value={deviceType.slug}>
              {deviceType.name}
            </TabsTrigger>
          ))}
        </TabsList>
      </Tabs>

      {/* Question Groups Grid */}
      {isGroupsLoading || isDeviceTypesLoading ? (
        <div className="flex justify-center items-center h-64">
          <div className="animate-spin rounded-full h-12 w-12 border-t-2 border-b-2 border-blue-500"></div>
        </div>
      ) : filteredGroups && filteredGroups.length > 0 ? (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {filteredGroups.map((group) => (
            <Card
              key={group.id}
              className={`hover:shadow-md transition-shadow cursor-pointer ${!group.active ? "opacity-70" : ""}`}
              onClick={() => handleGroupClick(group.id)}
            >
              <CardHeader className="pb-2">
                <div className="flex items-start justify-between">
                  <div className="flex items-center">
                    {getGroupIcon(group)}
                    <div className="ml-4">
                      <CardTitle className="text-xl">{group.name}</CardTitle>
                      {!group.active && (
                        <Badge
                          variant="outline"
                          className="bg-gray-100 text-gray-500 mt-1"
                        >
                          Inactive
                        </Badge>
                      )}
                    </div>
                  </div>
                  <ChevronRight className="h-5 w-5 text-gray-400" />
                </div>
              </CardHeader>
              <CardContent>
                <p className="text-gray-600 line-clamp-2">
                  {group.statement || "No description provided."}
                </p>
                <div className="flex items-center mt-3 text-sm text-gray-500">
                  <span className="font-medium">
                    {group.questionsCount || 0} Questions
                  </span>
                  <span className="mx-2">•</span>
                  <span>
                    {deviceTypes?.find((dt) => dt.id === group.deviceTypeId)
                      ?.name || "No device type"}
                  </span>
                </div>
              </CardContent>
            </Card>
          ))}

          {/* Add New Group Card */}
          <Card
            className="border-dashed border-2 hover:border-blue-500 hover:shadow-sm transition-all cursor-pointer flex flex-col items-center justify-center h-[200px]"
            onClick={handleCreateGroup}
          >
            <CardContent className="flex flex-col items-center justify-center h-full">
              <Plus className="h-12 w-12 text-gray-400 mb-2" />
              <p className="text-gray-600 font-medium">
                Create New Question Group
              </p>
            </CardContent>
          </Card>
        </div>
      ) : (
        <div className="text-center py-12 border rounded-md bg-gray-50">
          <AlertCircle className="mx-auto h-12 w-12 text-gray-400" />
          <h3 className="mt-2 text-lg font-medium text-gray-900">
            No question groups found
          </h3>
          <p className="mt-1 text-gray-500">
            {searchQuery
              ? "No results match your search criteria."
              : "Get started by creating a new question group."}
          </p>
          <div className="mt-6">
            <Button onClick={handleCreateGroup}>
              <Plus className="mr-2 h-4 w-4" /> Create First Group
            </Button>
          </div>
        </div>
      )}
    </div>
  );
}
